% This TeX file provides authors with a basic template for generating manuscripts w/ the appropriate configurations/formatting for IEEE Transactions submissions. IEEE distributes the 'IEEEtran.cls' document class containing definitions of IEEE specific 'IEEE<x>' control sequences (CS) and environments for inserting many of the most common LaTeX2e macros/environments '<x>' (e.g. IEEEkeywords, IEEEeqnarray, etc.) w/ the appropriate IEEE Transactions manuscript formatting/appearance (for a detailed description of the 'IEEEtran.cls' document class and its usage, see http://texdoc.net/texmf-dist/doc/latex/IEEEtran/IEEEtran_HOWTO.pdf). The 'IEEEtrans-tools' package provided w/ this template configures useful third-party package macros/environments for IEEE submissions and provides additional tools that simplify the generation of IEEE manuscripts and other associated documents arising during the editing and review/resubmission process.
%
% Authors should begin by saving this template with the desired .tex file name and change the values of the example 'key=value' titling, authorship, and abstract/keywords options of the 'IEEEdocument' environment to match their own manuscript information (author affiliations and/or abstract/keywords may be omitted, but this is not recommended). This simplifies the process of specifying manuscript information and automatically enforces the appropriate formatting for the chosen 'doctype', incl. the standard IEEE formatting of the 1st word of the 1st \section ("Introduction" by default, optionally renamed w/ the 'introtitle' option). Additional control sequences an author wishes to use within their particular document may be defined in the corresponding "Document specific author macro definitions" section below.
%
% Authors should then set the 'IEEEtrans-tools' package options to specify (1) any external documents w/ content they wish to \ref reference, (2) various formatting options, and (3) the desired type of document to be generated (manuscript, markedup, preprint, arxiv). Sidebar, watermarking, and editing options only affect document formatting/appearance if they're appropriate for the selected 'doctype'; e.g. sidebars are only displayed if 'doctype=markedup', watermarking and the required IEEE submission note only appear if 'doctype=preprint/arxiv', and all watermarking/editing is suppressed if 'doctype=markedup/manuscript'. Hence, authors can use the same .tex file to generate each type of document by simply specifying the corresponding 'doctype' without (1) removing any sidebars inserted to mark modified text or (2) changing/removing any of the watermarking/sidebar/editing options.
%

    \begin{algorithmic}[1]%\label{alg:NTVS} % The number tells where the line numbering should start
    \State set $k = 0$
    \State set $\ell_{-1} = 0$
    \State set $x^k = \bar{x}$
    \While{$k<K\lparen$}
    \State set $n = 0$
    \State set $\ell_{k} = \textrm{rand}(k,\ell_{k-1})$\label{alg:lk}
    \State set $x^{k,n} = x^k$
    \While{$\boldsymbol{n < N}$}\label{alg:N}
            \State set $v^{k,n}=\phi'(x^{k,n})$
            \State set $\beta_{k,n} = \alpha^{\ell_{k}}$\label{alg:beta}
            \State set $x^{k,n+1} = x^{k,n} + \beta_{k,n}v^{k,n}$
            \State set $n = n + 1$
            \State set $\ell_{k} = \ell_{k} + 1$
        \EndWhile
        \State set $x^{k+1} = P_T(x^{k,N})$
        \State set $k = k + 1$
    \EndWhile
    \end{algorithmic}\label{alg:NTVS}
